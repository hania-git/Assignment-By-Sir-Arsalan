// 12.   Create a link that opens a new window when the user clicks on it and then closes the window when the user clicks on a close button in the new window. The size of the new window should be determined by the user's mouse position.

document.addEventListener('DOMContentLoaded', () => {
    var openLink = document.getElementById('openLinknow');

    openLink.addEventListener('click', (event) => {
        event.preventDefault();

        var mouseX = event.clientX;
        var mouseY = event.clientY;
        var windowWidth = Math.min(800, window.innerWidth - mouseX);
        var windowHeight = Math.min(600, window.innerHeight - mouseY);
        var newWindow = window.open('', '', `width=${windowWidth},height=${windowHeight}`);
        newWindow.document.write(`
            <h1>New Window Content</h1>
            <p>This window was opened at (${mouseX}, ${mouseY}).</p>
            <button id="closeButton">Close Window</button>
        `);
        var closeButton = newWindow.document.getElementById('closeButton');
        closeButton.addEventListener('click', () => {
            newWindow.close(); 
        });
    });
});

// 6.   Create a guessing game where the user has to guess a number between 1 and 100. 
// while loop
var secretNumber = Math.floor(Math.random() * 100) + 1;
var guess;
var attempts = 0;

while (true) {
    guess = parseInt(prompt("Guess the number between 1 and 100:"));
    attempts++;

    if (isNaN(guess)) {
        alert("Invalid input. Please enter a valid number.");
    } else if (guess < 1 || guess > 100) {
        alert("Number out of range. Please enter a number between 1 and 100.");
    } else if (guess === secretNumber) {
        alert("Congratulations! You guessed the number in " + attempts + " attempts.");
        break;
    } else if (guess < secretNumber) {
        alert("Too low. Try again.");
    } else {
        alert("Too high. Try again.");
    }
}

// do-while loop
